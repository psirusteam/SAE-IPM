source("Modelo_bayes_HxA/0funciones/Estimar_ipm.R")
ipm_dam <- estime_IPM(poststrat = censo_ipm,
                      chain_ci = chain_ci,
                      chain_ind = chain_ind,
           byMap = "dam"
           ) %>% data.frame()
